<?php

    //connect to the database
    $db = mysqli_connect("localhost", "root", "", "BraingymmerDB");

    if (isset($_POST['btnUserRegister'])){
        session_start();
        $username = mysql_real_escape_string($_POST['username']);
        $email = mysql_real_escape_string($_POST['emailaddress']);
        $password = mysql_real_escape_string($_POST['password']);
        $retypePassword = mysql_real_escape_string($_POST['retypePassword']);
    
        //check if password was typed in correctly
        if ($password == $retypePassword){
        //create a user
            //hash the password before storing it in the database
            $password = md5($password);
            
            $sql = "INSERT INTO users(username, email, password) VALUES('$username', '$email', '$password')";

            mysqli_query( $db, $sql);
            $_SESSION['message'] = "you are now logged in";
            $_SESSION['username'] = $username;
            header("location: ../src/home.php"); //redirect to home page
        }else{
            $_SESSION['message'] = "Passwords do not match";
        }
    
    }

?>
